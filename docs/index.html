Basic usage, multiple sliders, labels.

new JSR({
    modules: [
        new JSR.Rail(),
        new JSR.Slider(),
        new JSR.Bar(),
        new JSR.Label(),
    ],
    config: {
        min: 0,
        max: 100,
        step: 1,
        initialValues: [25,75],
        container: document.getElementById('demo1'),
    }
});

Large values, limit (visible!), affix.

new JSR({
    modules: [
        new JSR.Rail(),
        new JSR.Slider(),
        new JSR.Bar(),
        new JSR.Label({
            formatter: (v) => `$${v}`
        }),
        new JSR.Limit(),
    ],
    config: {
        min: 10000,
        max: 20000,
        step: 1,
        initialValues: [15000, 17500],
        container: document.getElementById('demo2'),
        limit: {
            min: 12000,
            max: 16000,
        }
    }
});
